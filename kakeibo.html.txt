function compute(payments,income){
  const totalIncome=income.reduce((s,i)=>s+i.amount,0);
  const totalExpense=payments.reduce((s,p)=>s+p.amount,0);
  const totalPaid=payments.filter(p=>p.paid).reduce((s,p)=>s+p.amount,0);
  const totalUnpaid=payments.filter(p=>!p.paid).reduce((s,p)=>s+p.amount,0);
  const balance=totalIncome-totalExpense;
  const pct=totalExpense>0?Math.round(totalPaid/totalExpense*100):0;
  return{totalIncome,totalExpense,totalPaid,totalUnpaid,balance,pct};
}

function MonthlyChart({currentMonth,getMonthData}){
  const ref=useRef(null);
  const months=[];
  const[cy,cm]=currentMonth.split('-').map(Number);
  for(let i=5;i>=0;i--){const d=new Date(cy,cm-1-i,1);months.push(d.getFullYear()+'-'+String(d.getMonth()+1).padStart(2,'0'));}
  useEffect(()=>{
    const canvas=ref.current;if(!canvas)return;
    const ctx=canvas.getContext('2d');
    const dpr=window.devicePixelRatio||1;
    const W=canvas.parentElement.clientWidth;const H=160;
    canvas.width=W*dpr;canvas.height=H*dpr;canvas.style.width=W+'px';canvas.style.height=H+'px';
    ctx.scale(dpr,dpr);
    const incs=[],exps=[],bals=[];
    months.forEach(m=>{const{totalIncome,totalExpense,balance}=getMonthData(m);incs.push(totalIncome);exps.push(totalExpense);bals.push(balance);});
    const PAD={t:16,r:8,b:8,l:8};const cW=W-PAD.l-PAD.r;const cH=H-PAD.t-PAD.b;
    const n=months.length;const gap=cW/n;const bw=Math.floor(gap*0.26);
    const maxV=Math.max(...incs,...exps,1);
    const bx=(i,o)=>PAD.l+i*gap+gap/2+o;
    ctx.clearRect(0,0,W,H);
    [1,2,3].forEach(g=>{ctx.strokeStyle='rgba(255,255,255,0.04)';ctx.lineWidth=1;ctx.beginPath();ctx.moveTo(PAD.l,PAD.t+cH*(1-g/3));ctx.lineTo(PAD.l+cW,PAD.t+cH*(1-g/3));ctx.stroke();});
    months.forEach((_,i)=>{
      let bh=Math.max(2,(incs[i]/maxV)*cH);let by=PAD.t+cH-bh;
      const g1=ctx.createLinearGradient(0,by,0,by+bh);g1.addColorStop(0,'rgba(82,214,122,0.85)');g1.addColorStop(1,'rgba(82,214,122,0.1)');
      ctx.fillStyle=g1;ctx.beginPath();ctx.roundRect(bx(i,-bw-1),by,bw,bh,[3,3,0,0]);ctx.fill();
      bh=Math.max(2,(exps[i]/maxV)*cH);by=PAD.t+cH-bh;

      ctx.fillStyle=g2;ctx.beginPath();ctx.roundRect(bx(i,1),by,bw,bh,[3,3,0,0]);ctx.fill();
    });
    ctx.strokeStyle='rgba(91,143,249,0.9)';ctx.lineWidth=2;ctx.lineJoin='round';ctx.beginPath();
    months.forEach((_,i)=>{const lx=bx(i,0);const b=bals[i];const ly=b>=0?PAD.t+cH-(b/maxV)*cH:PAD.t+cH+Math.min((Math.abs(b)/maxV)*cH,cH*0.15);i===0?ctx.moveTo(lx,ly):ctx.lineTo(lx,ly);});
    ctx.stroke();
    months.forEach((_,i)=>{const lx=bx(i,0);const b=bals[i];const ly=b>=0?PAD.t+cH-(b/maxV)*cH:PAD.t+cH+Math.min((Math.abs(b)/maxV)*cH,cH*0.15);ctx.fillStyle=b>=0?C.green:C.red;ctx.beginPath();ctx.arc(lx,ly,3.5,0,Math.PI*2);ctx.fill();ctx.fillStyle=C.bg;ctx.beginPath();ctx.arc(lx,ly,1.5,0,Math.PI*2);ctx.fill();});

  const[masterInc,setMasterInc]=useState(()=>LS.get('kk_mi')||DEFAULT_MASTER_INC);
  const[payMap,setPayMap]=useState(()=>{const s=LS.get('kk_pm');return s||{'2025-06':JSON.parse(JSON.stringify(DEFAULT_PAY_0625))};});
  const[incMap,setIncMap]=useState(()=>{const s=LS.get('kk_im');return s||{'2025-06':JSON.parse(JSON.stringify(DEFAULT_INC_0625))};});

  useEffect(()=>{LS.set('kk_mp',masterPay);},[masterPay]);
  useEffect(()=>{LS.set('kk_mi',masterInc);},[masterInc]);
  useEffect(()=>{LS.set('kk_pm',payMap);},[payMap]);
  useEffect(()=>{LS.set('kk_im',incMap);},[incMap]);

  function getPay(m){return payMap[m]||(masterPay.map(p=>({...p,id:uid(),paid:false})));}
  function getInc(m){return incMap[m]||(masterInc.map(i=>({...i,id:uid()})));}
  function setPay(m,upd){setPayMap(prev=>{const cur=prev[m]||masterPay.map(p=>({...p,id:uid(),paid:false}));return{...prev,[m]:typeof upd==='function'?upd(cur):upd};});}
  function setInc(m,upd){setIncMap(prev=>{const cur=prev[m]||masterInc.map(i=>({...i,id:uid()}));return{...prev,[m]:typeof upd==='function'?upd(cur):upd};});}

  function changeMonth(dir){
    const[y,mo]=currentMonth.split('-').map(Number);
    const d=new Date(y,mo-1+dir,1);
    const nm=d.getFullYear()+'-'+String(d.getMonth()+1).padStart(2,'0');
    if(!payMap[nm])setPayMap(p=>({...p,[nm]:masterPay.map(x=>({...x,id:uid(),paid:false}))}));
    if(!incMap[nm])setIncMap(p=>({...p,[nm]:masterInc.map(x=>({...x,id:uid()}))}));
    setCurrentMonth(nm);
  }
  function applyMaster(){
    if(!window.confirm(`マスターを${mTitle(currentMonth)}に適用しますか？`))return;
    setPayMap(p=>({...p,[currentMonth]:masterPay.map(x=>({...x,id:uid(),paid:false}))}));
    setIncMap(p=>({...p,[currentMonth]:masterInc.map(x=>({...x,id:uid()}))}));
  }
  function resetMonth(){
    if(!window.confirm('支払い済み状態をリセットしますか？'))return;
    setPay(currentMonth,prev=>prev.map(p=>({...p,paid:false})));
  }
  function getMonthData(m){
    const pay=payMap[m]||masterPay.map(p=>({...p,paid:false}));
    const inc=incMap[m]||masterInc;
    return compute(pay,inc);
  }

  const payments=getPay(currentMonth);
  const income=getInc(currentMonth);
  const TABS=[{id:'dash',icon:'📊',label:'ダッシュ'},{id:'income',icon:'💰',label:'収入'},{id:'receipt',icon:'📷',label:'レシート',sp:true},{id:'pay',icon:'💳',label:'支払い'},{id:'settings',icon:'⚙️',label:'設定'}];

  return(
    <div style={S.app}>
      {page==='dash'&&<><div style={S.hdr}><div style={S.hdrSub}>家計管理</div><div style={S.hdrRow}><div style={S.hdrTitle}>{mTitle(currentMonth)}</div><div style={{display:'flex',gap:4}}><button onClick={()=>changeMonth(-1)} style={S.mnBtn}>‹</button><button onClick={()=>changeMonth(1)} style={S.mnBtn}>›</button></div></div></div><DashPage payments={payments} income={income} currentMonth={currentMonth} getMonthData={getMonthData}/></>}
      {page==='income'&&<><div style={S.hdr}><div style={S.hdrSub}>収入</div><div style={S.hdrRow}><div style={S.hdrTitle}>{mTitle(currentMonth)}</div><div style={{display:'flex',gap:4}}><button onClick={()=>changeMonth(-1)} style={S.mnBtn}>‹</button><button onClick={()=>changeMonth(1)} style={S.mnBtn}>›</button></div></div></div><IncomePage income={income} setIncome={upd=>setInc(currentMonth,upd)}/></>}
      {page==='receipt'&&<><div style={S.hdr}><div style={S.hdrSub}>レシート読み取り</div><div style={S.hdrRow}><div style={S.hdrTitle}>📷 レシート</div><div style={{fontSize:11,color:C.muted}}>{mTitle(currentMonth)}</div></div></div><ReceiptPage addPayments={items=>setPay(currentMonth,prev=>[...prev,...items])}/></>}
      {page==='pay'&&<><div style={S.hdr}><div style={S.hdrSub}>支払い</div><div style={S.hdrRow}><div style={S.hdrTitle}>{mTitle(currentMonth)}</div><div style={{display:'flex',gap:4}}><button onClick={()=>changeMonth(-1)} style={S.mnBtn}>‹</button><button onClick={()=>changeMonth(1)} style={S.mnBtn}>›</button></div></div></div><PayPage payments={payments} setPayments={upd=>setPay(currentMonth,upd)}/></>}
      {page==='settings'&&<><div style={S.hdr}><div style={S.hdrSub}>設定</div><div style={S.hdrRow}><div style={S.hdrTitle}>マスター設定</div></div></div><SettingsPage masterPay={masterPay} masterInc={masterInc} setMasterPay={setMasterPay} setMasterInc={setMasterInc} applyMaster={applyMaster} resetMonth={resetMonth}/></>}
      <div style={S.tabs}>
        {TABS.map(t=>(
          <button key={t.id} onClick={()=>setPage(t.id)} style={S.tab(page===t.id)}>
            {t.sp?<div style={{width:34,height:34,borderRadius:99,background:`linear-gradient(135deg,${C.orange},${C.red})`,display:'flex',alignItems:'center',justifyContent:'center',fontSize:17,marginTop:-6,boxShadow:'0 3px 10px rgba(249,168,77,0.35)'}}>{t.icon}</div>:<span style={{fontSize:19,lineHeight:1}}>{t.icon}</span>}
            {t.label}
          </button>
        ))}
      </div>
    </div>
  );
}

ReactDOM.createRoot(document.getElementById('root')).render(<App/>);
</script>
